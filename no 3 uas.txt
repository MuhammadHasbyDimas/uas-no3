<script>
function hitungGaji() {
 let gajiPokok = parseInt(document.getElementById("gajiPokok").value);
 let jamLembur = parseInt(document.getElementById("jamLembur").value);
 let gajiLembur = jamLembur * 20000;
 let gajiKotor = gajiPokok + gajiLembur;
 let pajak;
 if (gajiKotor < 5000000) {
 pajak = gajiKotor * 0.05;
 } else if (gajiKotor <= 10000000) {
 pajak = gajiKotor * 0.10;
 } else {
 pajak = gajiKotor * 0.15;
 }
 let gajiBersih = gajiKotor - pajak;
 document.getElementById("hasil").innerHTML =
 "Gaji Pokok: Rp " + gajiPokok.toLocaleString() + "<br><br>" +
 "Total Gaji Lembur: Rp " + gajiLembur.toLocaleString() + "<br><br>" +
 "Gaji Kotor: Rp " + gajiKotor.toLocaleString() + "<br><br>" +
 "Pajak: Rp " + pajak.toLocaleString() + "<br><br>" +
 "Gaji Bersih: Rp " + gajiBersih.toLocaleString();
}
</script>
